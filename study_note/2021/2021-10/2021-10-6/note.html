module top_module(
    input clk,
    input in,
    input reset,    // Synchronous reset
    output done
); 
    parameter IDLE = 2'b00, DATA = 2'b01, STOP = 2'b10, ERR = 2'b11;
    logic [1:0] state, next_state;
    logic [7:0] count;
    
    always_ff @(posedge clk) begin
        if (reset) state <= IDLE;
        else state <= next_state;
    end
    
    always_ff @(posedge clk) begin
        if (reset) count <= 0;
        else if (state == DATA) count <= count + 1'b1;
        else count <= 0;
    end
    
    always_comb begin
        case (state)
            IDLE: next_state = in ? IDLE : DATA;
            DATA: next_state = (count == 8)? (in? STOP : ERR) : DATA; 
            STOP: next_state = in? IDLE : DATA;
            ERR: next_state = in? IDLE : ERR;
        endcase
    end
    
    assign done = state == STOP;

endmodule

module top_module(
    input clk,
    input in,
    input reset,    // Synchronous reset
    output [7:0] out_byte,
    output done
); 
    parameter IDLE = 2'b00, DATA = 2'b01, STOP = 2'b10, ERR = 2'b11;
    logic [1:0] state, next_state;
    logic [7:0] count;
    logic [7:0] out_byte_r;
    
    always_ff @(posedge clk) begin
        if (reset) state <= IDLE;
        else state <= next_state;
    end
    
    always_ff @(posedge clk) begin
        if (reset) count <= 0;
        else if (state == DATA) count <= count + 1'b1;
        else count <= 0;
    end
    
    always_ff @(posedge clk) begin
        if (reset) out_byte_r <= 8'b0000_0000;
        else if (state == DATA & count == 0) out_byte_r <= {out_byte_r[7:1], in};
        else if (state == DATA & count == 1) out_byte_r <= {out_byte_r[7:2], in, out_byte_r[0]};
        else if (state == DATA & count == 2) out_byte_r <= {out_byte_r[7:3], in, out_byte_r[1:0]};
        else if (state == DATA & count == 3) out_byte_r <= {out_byte_r[7:4], in, out_byte_r[2:0]};
        else if (state == DATA & count == 4) out_byte_r <= {out_byte_r[7:5], in, out_byte_r[3:0]};
        else if (state == DATA & count == 5) out_byte_r <= {out_byte_r[7:6], in, out_byte_r[4:0]};
        else if (state == DATA & count == 6) out_byte_r <= {out_byte_r[7], in, out_byte_r[5:0]};
        else if (state == DATA & count == 7) out_byte_r <= {in, out_byte_r[6:0]};
        else out_byte_r <= out_byte_r;
    end
    
    always_comb begin
        case (state)
            IDLE: next_state = in ? IDLE : DATA;
            DATA: next_state = (count == 8)? (in? STOP : ERR) : DATA; 
            STOP: next_state = in? IDLE : DATA;
            ERR: next_state = in? IDLE : ERR;
        endcase
    end
    
    assign done = state == STOP;
    assign out_byte = out_byte_r;


endmodule

        else if (state == DATA & count == 8) out_byte_r <= {in, out_byte_r[7:0]};
        else out_byte_r <= out_byte_r;
    end
    
    always_comb begin
        case (state)
            IDLE: next_state = in ? IDLE : DATA;
            DATA: next_state = (count == 9)? (in? STOP : ERR) : DATA;

  1. 非组合型数组初始化用 '{} 对数组每一个维度进行赋值

拷贝

  1. 组合型数组赋值在左右两侧操作数的大小和维度不同时，也可以赋值，尺寸不当会截取或扩展右操作数。
  2. 非组合型数组要求左右两侧操作数的维度和大小必须严格一致。
  3. 非组合型数组和组合型数组无法互相直接赋值。

SV过程块和方法

initial 和 always

  1. module/endmodule, interface/endinterface 被视为硬件世界。program/endprogram, class/endclass被视为软件世界。
  2. always 用来描述硬件时序电路和组合电路，只可以用在 module 或者 interface 中。
  3. initial 符合软件执行方式，只执行一次。可以用在 module, interface, program 中使用。

软件方法-函数 function

  1. 可以在参数列表中指定输入参数input，输出参数output，输入输出参数inout，引用参数ref，默认参数为 input.
  2. 可以返回数值或者不返回数值void
function int double(input a);
    return 2*a;
endfunction
  1. function 默认数据类型为 logic 例如input [7:0] addr，addr 是 logic 类型的四值逻辑。
  2. 数组可以作为形式参数传递。
  3. function 有返回值需要关键词 return，没有返回值时应该在声明是采用 void function()
  4. 只有数据变量可以在形式参数列表中被声明为 ref，硬件信号不行。
  5. 使用 ref 时，可以通过 const 来限定 ref 声明的参数来保护数据对象只被读取不被写入。
  6. 声明参数时可以给入默认数值 input [7:0] addr = 0

软件方法-任务 task

  1. task 无法通过 return 返回结果，只能通过 output, inout, reg返回
  2. task 内可以置入耗时语句，function 不行，耗时语句包括 @event, wait event, #delay等
task mytask1 (output logic [31:0] x, input logic y);
    ...
endtask
  1. function 和 task 均可调用 function，只有 task 可以调用 task.

软件方法-变量生命周期

  1. 生命周期分为动态 automatic 和 静态 static。
  2. 局部变量的生命周期是动态的，伴随其所在域，例如 function/task 中的局部变量，方法调用结束后，变量的生命周期也就终结。
  3. 全局变量从程序执行开始到结束一直存在，例如 module 中的变量默认为全局变量，硬件信号全部都是静态生命周期。
  4. 如果某数据变量被声明为 automatic，那么在进入该进程/方法后，该变量会被创建，离开该进程/方法后，该变量会被销毁。
  5. static 变量在仿真开始时被创建，可以被多个进程和方法共享。
module test();
    function automatic int auto_cnt(input a); // 该函数中所有变量都为automatic
        int cnt = 0;
        cnt += a;
        return cnt;
    endfunction

    function static int static_cnt(input a); // 该函数中所有变量都为static
        static int cnt = 0;
        cnt += a;
        return cnt;
    endfunction

    function int def_cnt(input a); // 默认函数都是static的
        static int cnt = 0; //static写不写没有区别
        cnt += a;
        return cnt;
    endfunction

    initial begin
        $display ("@1 auto_cnt = %0d", auto_cnt(1)); // 输出结果：@1 auto_cnt = 1
        $display ("@2 auto_cnt = %0d", auto_cnt(1)); // 输出结果：@2 auto_cnt = 1 
        $display ("@1 static_cnt = %0d", static_cnt(1)); // 输出结果：@1 static_cnt = 1
        $display ("@2 static_cnt = %0d", static_cnt(1)); // 输出结果：@2 static_cnt = 2
        $display ("@1 def_cnt = %0d", def_cnt(1)); // 输出结果：@1 def_cnt = 1
        $display ("@2 def_cnt = %0d", def_cnt(1)); // 输出结果：@2 def_cnt = 2
    end
endmodule
  1. 对于 automatic 或者 static 方法，可以对内部定义的变量做单个声明。
  2. 在module, program, interface, task, function之外声明的变量拥有静态生命周期。
  3. 在module, interface, program 内部声明，且在 task, process, function 外部声明的变量也具有静态生命周期，作用域在该块中。
  4. 在module, program, interface 中定义的task, function默认都是静态方法。
  5. 在task, function, process 中定义的变量，其类型默认跟随过程块的类型，但可以自定义显式声明。

注意分析COUNT的状态转换

module top_module(
    input clk,
    input reset,    // Synchronous reset
    input in,
    output disc,
    output flag,
    output err);

    parameter IDLE = 1, COUNT = 2, DISC = 3, FLAG = 4, ERROR = 5;
    logic [2:0] state, next_state;
    logic [7:0] count;
    
    always_ff @(posedge clk) begin
        if (reset) state <= IDLE;
        else state <= next_state;
    end
    
    always_ff @(posedge clk) begin
        if (reset) count <= 8'b0000_0000;
        else if (state == COUNT) count <= count + 1'b1;
        else count <= 0;
    end
    
    always_comb begin
        case(state)
            IDLE: if (!in) next_state = IDLE;
            	else next_state = COUNT;
            COUNT: if (count < 4 & !in) next_state = IDLE;
            else if (count == 4 & !in) next_state = DISC;
            else if (count == 5 & !in) next_state = FLAG;
            else if (count == 5 & in) next_state = ERROR;
            	else next_state = COUNT;
            DISC: if (in) next_state = COUNT;
            	else next_state = IDLE;
            FLAG: if (in) next_state = COUNT;
            	else next_state = IDLE;
            ERROR: if (in) next_state = ERROR;
            	else next_state = IDLE;
        endcase
    end
    
    assign disc = state == DISC;
    assign flag = state == FLAG;
    assign err = state == ERROR;
endmodule

首先0保持不变，第一个1保持不变，之后每一位都取反。

module top_module (
    input clk,
    input areset,
    input x,
    output z
); 
    
    parameter S1 = 0, S2 = 1, S3 = 2;
    logic [1:0] state, next_state;
    
    always_ff @(posedge clk or posedge areset) begin
        if (areset) state <= S1;
        else state <= next_state;
    end
    
    always_comb begin
        case(state)
        	S1: next_state = x? S2:S1;
            S2: next_state = x? S3:S2;
            S3: next_state = x? S3:S2;
        endcase
    end
    
    assign z = state == S2;

endmodule
