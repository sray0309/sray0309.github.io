	  forever chnl1_init.chnl_write(chnl1_gen.get_data());
	  forever chnl2_init.chnl_write(chnl2_gen.get_data());
    join_none

    $display("fifo_full_test: stop 3 initiators running");
    disable fork_all_run;

      chnl0_init.chnl_idle();

  logic [31:0] ch_data;

  clocking drv_ck @(posedge clk);

  class chnl_initiator;
    local string name;
    local int idle_cycles;
    local virtual chnl_intf intf;
  
    function new(string name = "chnl_initiator");
      this.name = name;
      this.idle_cycles = 1;
    endfunction
  
    function void set_idle_cycles(int n);
      this.idle_cycles = n;
    endfunction
  
    function void set_name(string s);
      this.name = s;
    endfunction
  
    function void set_interface(virtual chnl_intf intf);
      if(intf == null)
        $error("interface handle is NULL, please check if target interface has been intantiated");
      else
        this.intf = intf;
    endfunction
  
    task chnl_write(input chnl_trans t); //发送一个数据包
      @(posedge intf.clk);
      intf.drv_ck.ch_valid <= 1;
      intf.drv_ck.ch_data <= t.data;
      wait(intf.ch_ready === 'b1);
      $display("%t channel initiator [%s] sent data %x", $time, name, t.data);
      repeat(idle_cycles) chnl_idle();
    endtask
    
    task chnl_idle();
      @(posedge intf.clk);
      intf.drv_ck.ch_valid <= 0;
      intf.drv_ck.ch_data <= 0;
    endtask
  endclass: chnl_initiator
  1. 声明generator类
  class chnl_generator;
    chnl_trans trans[$];
    int num;
    int id;
    function new(int n);
      this.id = n;
      this.num = 0;
    endfunction
    function chnl_trans get_trans(); //返回一个数据包
      chnl_trans t = new();
      t.data = 'h00C0_0000 + (this.id<<16) + this.num;
      t.id = this.id;
      t.num = this.num;
      this.num++;
      this.trans.push_back(t);
      return t;
    endfunction
  endclass: chnl_generator

foreach 和 fork...join配合使用

fork
    agent[0].run();
    agent[1].run();
    agent[2].run();
join

上述代码可以用foreach循环来简化

  1. 使用 fork begin ... end join 来隔离线程，防止 wait join 对之前的线程产生影响。
  2. 使用内部定义变量来代替foreach的循环变量，因为每个独立线程需要独立变量，而foreach中的循环变量是同一个
  3. 利用 fork ... join_none 来实现每个循环中的函数同时并行运行。
  4. 利用 wait fork; 来等待前面所有并行线程的结束，实现 fork ... join 的效果
fork begin
    foreach (agent[i]) begin
        automatic int j = i;
        fork
          agent[j].run();
        join_none
    end
    wait fork;
end
join

注意这个 begin ... end 非常重要，没有它会使 wait fork; 成为一个独立的线程，无法控制之前的 fork ... join_none

SystemVerilog 随机与约束

随机约束与分布

  1. 随机内容
    • 器件配置
    • 环境配置
    • 原始输入数据
    • 延时
    • 协议异常
  2. 声明随机变量的类
    • rand 表明随机属性，只有类的成员变量可以随机
    • randc 表示周期随机性，表示所有可能的值都赋过随机值后才可能重复
    • 随机属性需要配合预定义的类随机函数 std::randomize() 使用，只有通过声明 rand 变量，在后期可以通过对象调用 randomize() 函数才可以随机化变量
    • 约束 constraint 也同随机变量一起在类中声明
class Packet;
    rand bit [31:0] src, dst, data[8];
    randc bit [7:0] kind;

    constraint c {
        src > 10;
        src < 15;
    }
endclass

Packet p;

initial begin
    p = new();
    assert (p.randomize()) else $fatal(0, "Packet::randomize failed");
end
  1. SV只能随机出两值逻辑，随机不出来x和z
class data;
    rand bit [2:0] month;
    rand bit [4:0] day;
    rand int year;
    constraint c_date {
        month inside {[1:12]}; //注意month的位宽，最大值为7，随机不出来大于7的数
        day inside {[1:31]};
        year inside {[2010:2030]}; 
    }
endclass

class Stim;
    const bit [31:0] CONGEST_ADDR = 42;

    typedef enum {READ, WRITE, CONTROL} stim_e;
    randc stim_e kind;
    rand bit [31:0] len, src, dst;
    bit congestion_test;

    constraint c_stim {
        len < 1000;
        len > 0;
        if (congestion_test) {
            dst insede {[CONGEST_ADDR-100:CONGEST_ADDR+100]};
            src == CONGEST_ADDR;
        }
        else
        src inside {0, [2:10], [100:107]};
    }
endclass
  1. 权重分布
    • dist 在约束中用来产生随机数值的权重分布，值或权重可以是常数或者变量
    • 权重不用百分比表示，权重的和也可以不是100
    • := 表示范围内每一个值的权重是相同的，:/ 表示权重要平均分到值范围内的每一个值
rand int src, dst;
constraint c_dist {
    src dist {0:=40, [1:3]:=60};
    // src = 0, weight = 40/220
    // src = 1, weight = 60/220
    // src = 2, weight = 60/220
    // src = 3, weight = 60/220
    dst dist {0:/40, [1:3]:/60};
    // dst = 0, weight = 40/100
    // dst = 1, weight = 20/100
    // dst = 2, weight = 20/100
    // dst = 3, weight = 20/100
}
  1. inside 运算符
    • 表示变量应该属于某一个值的集合，在集合里取值的概率是相等的
    • 可以使用 $ 指定最大值和最小值
  2. 条件约束
    • 可以通过 -> 让一个约束表达式在特定时刻有效
    constraint c_io {
        (io_space_mode) -> addr[31] == 1'b1;
    }
  1. 双向约束
    • 约束块不是自上向下的程序性代码，是并行的，所有的约束表达式同时有效
    • 约束是双向的，所有随机变量的约束会同时计算，增加或删除任何一个变量的约束都会直接或间接影响所有相关的值的选取

约束快控制

  1. 打开或关闭约束
    • 一个类可以包含多个约束块，一般各个约束块之间的约束内容是互相协调不违背的，其他情况可根据需要选择使能哪些约束快，禁止哪些约束快。
    • 使用 constraint_mode() 函数打开或者关闭约束
class Packet;
    rand int length
    constraint c_short {
        lenth inside {[1:31]};
    }
    constraint c_long {
        length inside {[1000:1023]};
    }
endclass

Packet p;
initial begin
    p = new();
    p.c_short.constraint_mode(0);// create long packet
    assert (p.randomize());
    transmit(p);

    p.constraint_mode(0);
    p.c_short.constraint_mode(1);// create short packet
    assert (p.randomize());
    transmit(p);
end
  1. 内嵌约束
    • SV允许使用 randomize() with 来增加额外的约束，要注意内部约束和外部约束之间应该是协调的。
    • 软约束 soft 可以允许当外部或其他约束与当前约束有冲突时，替换掉当前约束
class Transaction;
    rand bit [31:0] addr, data;
    constraint c1 {
        soft addr inside {[0:100], [1000:2000];}
    }
endclass

Transaction t;

initial begin
    t = new();
    assert (t.randomize() with {addr >= 50; addr <= 1500; data < 10;});

    draveBus(t);

    assert (t.randomize() with {addr == 2000; data > 10;});
    driveBus(t);
end

随机函数

  1. pre_randomize() 和 post_randomize()
    • 可以在类中定义上述两个函数，分别在其中定义随机化前的行为和随机化后的行为。对象在执行 randomize() 之前或之后会分别执行这两个函数。
    • 这两个函数可以看做是 randomize() 函数的回调函数
  2. 随机数函数
    • $random() 平均分布，返回32位有符号随机数
    • $urandom() 平均分布，返回32位无符号随机数
    • $urandom_range() 在指定范围内的平均分布
  3. 随机化个别变量
    • 在调用 randomize() 时可以传递变量的一个子集，这样只会随机化类里的几个变量
    • 只有参数列表里的变量才会被随机化，其他变量会被当作状态变量而不会被随机化
    • 所有的约束仍然保持有效
    • 类里所有被指定或者没有被指定rand的变量都可以作为 randomize() 的参数而被随机化
class Rising;
    byte low;
    rand byte med, hi;
    constraint up {
        low < med;
        med < hi;
    }
endclass

initial begin
    Rising r;
    r = new();
    r.randomize(); // 随机化hi，但不改变low
    r.randomize(med); // 只随机化med
    r.randomize(low); // 只随机化low
end

上述代码如果在initial中只做 r.randomize(low) 会报错，因为med和hi的初始值都为0，不满足 med < hi 的约束条件，因此整个随机会报错。
